37 //num is the number of sys call
38 //0x0 .. 0xfff - must be called in SVC/IRQ context
39 //0x1000 .. 0x1fff - must be called in SYS/SVC/IRQ context
40 //0x2000 .. - can be called in any context
41 
42 #define CALL_GROUP 0x10
43 #define CALL_GROUP_MASK 0xfffffff0
44 #define CALL_CONTEXT 0x1000

46 typedef enum {
47  //supervisor context
48  SYS_CALL_THREAD = 0x0 * CALL_GROUP,
49  SYS_CALL_MUTEX = 0x1 * CALL_GROUP,
50  SYS_CALL_EVENT = 0x2 * CALL_GROUP,
51  SYS_CALL_SEMAPHORE= 0x3 * CALL_GROUP,
52  SYS_CALL_QUEUE = 0x4 * CALL_GROUP,
53  SYS_CALL_SYS_TIMER= 0x5 * CALL_GROUP,
54  //system context
55  SYS_CALL_TIME = CALL_CONTEXT + 0x0 * CALL_GROUP,
56  SYS_CALL_MEM = CALL_CONTEXT + 0x1 * CALL_GROUP,
57  SYS_CALL_DBG = CALL_CONTEXT + 0x2 * CALL_GROUP
58 }SYS_CALL_SUPERVISOR;
